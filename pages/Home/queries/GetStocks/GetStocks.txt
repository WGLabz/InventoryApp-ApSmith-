SELECT c.name,
c.component_id,
c.description,
c.date_added,
c.image_content,
c.image_type,
b.box_code,
c.category_id,
b.box_label,
ct.category_name,
b.box_id,
( SELECT SUM(quantity_ordered) FROM order_parts WHERE component_id = c.component_id) AS purchased,
COALESCE((SELECT SUM(quantity_used) FROM project_components WHERE component_id = c.component_id) , 0) AS used,
(( SELECT SUM(quantity_ordered) FROM order_parts WHERE component_id = c.component_id) -
COALESCE((SELECT SUM(quantity_used) FROM project_components WHERE component_id = c.component_id) , 0) ) AS stock
FROM components c
LEFT 
JOIN boxes b ON b.box_id = c.box_id
LEFT
JOIN categories ct ON ct.category_id = c.category_id
WHERE c.name LIKE '%{{ ComponentsTable.searchText}}%'
ORDER BY {{!!ComponentsTable.sortOrder.column ? ComponentsTable.sortOrder.column : 'name'}} {{!!ComponentsTable.sortOrder.order ? !!ComponentsTable.sortOrder.order : 'desc'}}
LIMIT {{ComponentsTable.pageSize}} OFFSET {{(ComponentsTable.pageNo - 1) * ComponentsTable.pageSize}};
